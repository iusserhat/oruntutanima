💡
LabelEncoder Nasıl Çalışır?

Sınıfları alfabetik sıralar ve 0'dan başlayarak sayı verir.

  • ['F', 'M'] → F=0, M=1
  • ['HIGH', 'LOW', 'NORMAL'] → HIGH=0, LOW=1, NORMAL=2
  • ['No', 'Yes'] → No=0, Yes=1

📈 Simple Linear Regression

from sklearn.linear_model import LinearRegression

# Feature ve label hazırla (numpy array)
train_x = np.asanyarray(train[['ENGINESIZE']])    # [n,1] şeklinde 2D
train_y = np.asanyarray(train[['CO2EMISSIONS']])

regr = LinearRegression()

📉 Non-Linear Regression / Sigmoid

import numpy as np
from scipy.optimize import curve_fit

# ─── Sigmoid Fonksiyonu Tanımı ───
def sigmoid(x, beta1, beta2):   # 'veri' parametresi olarak x
    y = 1 / (1 + np.exp(-beta1 * (x - beta2)))
    return y

# ─── İlk Tahminle Test Et ───
beta1 = 0.3
beta2 = 1990
y_pred = sigmoid(x, beta1, beta2)

# ─── Veriyi normalize et (0-1 arası) ───
xdata = x / max(x)           # 1, 2, 3, 30, 40 → 1/40, 2/40...
ydata = y / max(y)

# ─── En iyi beta1 ve beta2'yi bul (curve_fit) ───
popt, pcov = curve_fit(sigmoid, xdata, ydata)
# popt = [beta1_optimal, beta2_optimal]

print("beta1:", popt[0], "beta2:", popt[1])

# ─── Tahmin ───
y_pred_opt = sigmoid(xdata, *popt)   # *popt → popt listesini parametrelere yayar
# *popt yerine popt[0], popt[1] da yazılabilir

# ─── Accuracy ───
print("R² Accuracy:", r2_score(ydata, y_pred_opt))

# ─── Genel Non-Linear ─── (Quadratic)
x_q = np.arange(-5, 5, 0.1)      # np.arange(-5, 5, 0.1)
y_q = x_q**3 + x_q**2 + x_q + 3  # y = x³+x²+x+3
# Quadratic yazımı: np.power(x, 2)

🔮 KNN Sınıflandırma

from sklearn.neighbors import KNeighborsClassifier
from sklearn import preprocessing, metrics
import numpy as np

# 1. Normalizasyon (KNN için ZORUNLU)
X = preprocessing.StandardScaler().fit_transform(X.astype(float))

# 2. Train/Test
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.2, random_state=4
)

# 3. En iyi K'yı bul (1'den 10'a dene)
Ks = 10
mean_acc = np.zeros((Ks-1,))
for n in range(1, Ks):
    neigh = KNeighborsClassifier(n_neighbors=n).fit(X_train, y_train)
    yhat  = neigh.predict(X_test)
    mean_acc[n-1] = metrics.accuracy_score(y_test, yhat)

print('Best accuracy:', mean_acc.max(), 'at k=', mean_acc.argmax()+1)

# 4. Final model (en iyi k ile)
k = mean_acc.argmax() + 1
neigh = KNeighborsClassifier(n_neighbors=k)
neigh.fit(X_train, y_train)

# 5. Tahmin
yhat = neigh.predict(X_test)

# 6. Değerlendirme
print('accuracy:', metrics.accuracy_score(y_test, yhat))
print('jaccard:',  jaccard_score(y_test, yhat))
print('f1:',       f1_score(y_test, yhat, average='weighted'))

🌳 Decision Tree

from sklearn.tree import DecisionTreeClassifier, export_graphviz
from sklearn import preprocessing, metrics

# 1. LabelEncoder — Kategorik verileri sayıya çevir
le = preprocessing.LabelEncoder()
X['Sex']        = le.fit_transform(X['Sex'])        # M/F → 1/0
X['BP']         = le.fit_transform(X['BP'])         # LOW/NORMAL/HIGH → 0/1/2
X['Cholesterol'] = le.fit_transform(X['Cholesterol']) # NORMAL/HIGH → 0/1

# 2. Train/Test Split
X_train, X_test, y_train, y_test = train_test_split(
    X, y, test_size=0.3, random_state=3
)

# 3. Model oluştur ve eğit
drugTree = DecisionTreeClassifier(
    criterion='entropy',   # 'gini' veya 'entropy'
    max_depth=4            # Overfitting önlemi
)
drugTree.fit(X_train, y_train)

# 4. Tahmin ve değerlendirme
predTree = drugTree.predict(X_test)
print("Accuracy:", metrics.accuracy_score(y_test, predTree))

# 5. Görselleştirme (ağaçı çizdir)
import pydotplus, io
from sklearn.tree import export_graphviz

dot_data = io.StringIO()
export_graphviz(
    drugTree, out_file=dot_data,
    feature_names=['Age', 'Sex', 'BP', 'Cholesterol', 'Na_to_K'],
    class_names=drugTree.classes_,
    filled=True, special_characters=True
)
graph = pydotplus.graph_from_dot_data(dot_data.getvalue())
graph.write_png('drug_tree.png')
